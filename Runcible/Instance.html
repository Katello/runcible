def initialize(config = {})
  @config = {
    :api_path   => '/pulp/api/v2/',
    :url        => 'https://localhost',
    :user       => '',
    :http_auth  => {:password => {} },
    :headers    => {:content_type => 'application/json',
                    :accept       => 'application/json'},
    :logging    => {}
  }.merge(config).with_indifferent_access

  initialize_wrappers
end
